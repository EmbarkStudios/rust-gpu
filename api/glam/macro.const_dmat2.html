Creates a DMat2 from two column vectors that can be used to initialize a constant value.

use glam::{const_dmat2, DMat2};
const ZERO: DMat2 = const_dmat2!([0.0; 4]);
const IDENTITY: DMat2 = const_dmat2!([1.0, 0.0], [0.0, 1.0]);
